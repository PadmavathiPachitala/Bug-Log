Immediately delete the Redis session key on logout, and add the token's remaining TTL to a denylist set so in-flight tokens are also blocked:

            

              

                javascript
                
              

              
// auth.js — logout handler
router.post('/logout', authenticate, async (req, res) => {
  const { userId, jti, exp } = req.user;

  // 1. Delete the active session from Redis
  await redis.del(`session:user:${userId}`);

  // 2. Add token to denylist until its natural expiry
  const ttl = exp - Math.floor(Date.now() / 1000);
  if (ttl > 0) {
    await redis.setEx(`denylist:${jti}`, ttl, '1');
  }

  // 3. Update JWTMiddleware.verify to check denylist
  // if (await redis.exists(`denylist:${decoded.jti}`)) throw ...

  res.json({ success: true, message: 'Logged out' });
});
